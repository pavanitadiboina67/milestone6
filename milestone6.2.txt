for ( var i = 1; i <= 100; i++ )
{
  if ( i%3 === 0 && i%5 === 0 )
  {
    console.log( i + " talentiocompany" );
  }
  else if ( i%3 === 0 ) 
  {
    console.log(i+ " talentio" );
  }
  else if ( i%5 === 0 ) 
  {
    console.log(i+ " company" );
  }
  else
  {
    console.log(i);
  }
}
